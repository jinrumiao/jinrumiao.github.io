關於這份申請網頁，分享一個開發時的技術心得。

使用JS偵測class包含box標籤的物件，監聽scroll動作，當box物件的上邊界的位置小於螢幕畫面的五分之四時，就在box物件的classlist加上show標籤，並配合CSS讓box可以在觸發條件後出現在畫面的呈現中，當觸發條件失效則消失。因為有設定排序奇數與偶數，所以box物件出現時是一右一左當條件JS的條件觸發classlist加入show之後，讓box出現到螢幕畫面中，因為有設定排序奇數與偶數，所以box物件出現時是一右一左出現在畫面中。 

                        
// JS
const boxes = document.querySelectorAll('.box');  // 偵測class包含box標籤的物件

window.addEventListener('scroll', showBoxes);  // 搭配監聽scroll動作

function showBoxes() {
    const triggerBottom = (window.innerHeight / 5) * 4;  // 設定螢幕畫面的五分之四為觸發條件

    boxes.forEach((box) => {
        const boxTop = box.getBoundingClientRect().top;  // 取得每個box物件的上邊界位置

        // 當box物件的上邊界的位置小於螢幕畫面的五分之四時，就在box物件的classlist加上show標籤，否則移除box物件的show標籤
        if (boxTop < triggerBottom) { box.classList.add('show'); }
        else { box.classList.remove('show'); }
    });
}

/* CSS */

/* 設定基本box樣式 */
.box {
    background: white;
    width: 55%;
    /* height: 320px; */
    margin: 32px;
    border-radius: 0.25rem;
    box-shadow: 8px 8px 5px rgba(15, 36, 74, .6);
    display: grid;
    place-content: center;
    transform: translateX(400%) rotate(30deg);  /* 一開始先右移至畫面之外，並加上一30度的順時針旋轉 */
    transform-origin: center;
    transition: transform .6s ease-out;
}

/* 讓排序是偶數的box往左邊移動，並加上一30度的逆時針旋轉 */
.box:nth-of-type(even) {
    transform: translateX(-400%) rotate(-30deg)
}

/* 當條件JS的條件觸發classlist加入show之後，讓box出現到螢幕畫面中 */
.box.show {
    transform: translateX(0) rotate(0deg);
}
